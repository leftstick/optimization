风格之争


function get( obj, key, defaultVal )
{
  if(!obj){
    return defaultVal
  }
  return obj[key] || defaultVal
}

VS


const get = function(obj, key, defaultVal) {
    if(!obj){
        return defaultVal;
    }
    return obj[key] || defaultVal;
};

约定之争


const get = function(obj, key, defaultVal) {
    let res;
    if(obj) {
        if(obj[key]) {
            res = obj[key];
        }else {
            res = defaultVal;
        }
    } else {
        res = defaultVal;
    }
    return res;
};
