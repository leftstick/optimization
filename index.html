风格之争


function get( obj, key, defaultVal )
{
  if(!obj){
    return defaultVal
  }
  return obj[key] || defaultVal
}

VS


const get = function(obj, key, defaultVal) {
    if(!obj){
        return defaultVal;
    }
    return obj[key] || defaultVal;
};

约定之争


const get = function(obj, key, defaultVal) {
    let res;
    if(obj) {
        if(obj[key]) {
            res = obj[key];
        }else {
            res = defaultVal;
        }
    } else {
        res = defaultVal;
    }
    return res;
};

工程化问题赏析

太多请求

不使用CDN

未合理利用缓存

无压缩

太多跳转

太多DOM操作

使用过度资源，例如：大图片缩小显示

减少请求

  • 雪碧图、小图片转base64
  • CSS concatenation
  • JS concatenation, bundling

利用CDN

压缩静态资源

  • JavaScript minify
  • CSS minify
  • HTML minity
  • 网络请求gzip

不做无谓跳转

譬如：www.taobao.com/ => wwww.taobao.com

有SEO顾虑时，额外处理

减少DOM操作

  • 慎重思考大批量的DOM操作
  • 引入virtual DOM工具

使用合适的资源

  • 显示区域70px * 70px，就别用什么高清图
  • 只用个模版引擎，就别引入angular.js
  • 只用栅栏布局，就别把Bootstrap一整套都引入
  • 只用个“退出”图标，就别把一整套font-awesome引入

程序问题赏析

不使用严格模式


(function() {
    userName = 'HanMeimei';
}());
console.log('userName', userName);

改成


'use strict';
(function() {
    userName = 'HanMeimei';
}());
console.log('userName', userName);//ReferenceError

不用===


1 == '1'; // true
false == '  '; // true
true == [1]; // true

不清晰的表达式


if (options.minLat && options.maxLat && options.minLng && options.maxLng) {
    //...
}

改成


const hasCoordinate = options.minLat && options.maxLat && options.minLng && options.maxLng; 
if (hasCoordinate) {
    //...
}

执着使用for


var result = [];
for (var i = 0; i < homeIds.length; i++) {
    for (var j = 0; j < data.length; j++) {
        if (homeIds[i] == data[j].propertyId) {
            result.push(data[j]);
        }
    }
}

改成


const result = data.filter(d => homeIds.includes(d.propertyId));
